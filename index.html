Setup en 2 minutes :
  1. Ouvre Google Sheets → nouveau fichier
  2. Menu Extensions → Apps Script

  5. Copie l'URL générée → colle ci-dessus ✓
Les données seront envoyées dans 2 onglets séparés : Estimations et Ventes réelles.
function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var sheetName = data.type === 'estimation' ? 'Estimations' : 'Ventes réelles'; var sheet = ss.getSheetByName(sheetName); if (!sheet) sheet = ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) { var h = data.type === 'estimation' ? ['Date','Article','Achat','Vente est.','Livraison', 'Plateforme','Commission%','Commission€', 'Total frais','Profit estimé','Marge%','ROI%'] : ['Date','Article','Achat','Vente réelle','Livraison', 'Plateforme','Commission%','Commission€', 'Total frais','Profit réel','Marge%','ROI%','Lié à']; sheet.appendRow(h); } var row = [ data.date, data.item, data.buyPrice, data.sellPrice, data.shipping, data.platform, data.platformPct, data.platformFee, data.totalFees, data.profit, data.margin, data.roi ]; if (data.type === 'real') row.push(data.linkedEst || ''); sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
