function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var sheetName = data.type === 'estimation' ? 'Estimations' : 'Ventes réelles'; var sheet = ss.getSheetByName(sheetName); if (!sheet) sheet = ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) { var h = data.type === 'estimation' ? ['Date','Article','Achat/u','Qté','Achat total', 'Vente est./u','Livraison','Commission','Profit estimé'] : ['Date','Article','Achat/u','Qté','Achat total', 'Vente réelle/u','Livraison','Commission','Profit réel','Lié à']; sheet.appendRow(h); } var row = [ data.date, data.item, data.buyUnit, data.qty, data.buyTotal, data.sellUnit, data.shipping, data.commission, data.profit ]; if (data.type === 'real') row.push(data.linkedEst || ''); sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
